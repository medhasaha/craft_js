## Install

```bash
npm install --save hd-grapesjs
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'hd-grapesjs'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
